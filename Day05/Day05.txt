while문

while(조건식) { //조건식이 참일 경우 실행, 거짓일 경우 종료
	반복을 실행할 실행문;
}

//초기값과 증감식은 while문에서 필수 조건이 아님


do ~ while문

do{
	반복을 실행할 실행문;
}while(조건식) //조건식이 참일 경우 실행, 거짓일 경우 종료

//do ~ while문은 다른 반복문과는 다르게 조건식이 참이든 거짓이든
실행문을 무조건 한번 실행시킨 후 조건을 검사하여 반복을 결정한다


for문

for(초기값; 조건식; 증감식){ //조건식이 참일 경우 실행, 거짓일 경우 종료
	반복을 실행할 문장;
}
